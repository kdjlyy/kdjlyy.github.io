class Solution {
public:
    int n, m, path[202][202], res = 0;  // path[i][j] è¡¨ç¤º matrix[i][j] ä½ç½®çš„æœ€é•¿é€’å¢è·¯å¾„
    int d[5] = {-1, 0, 1, 0, -1};
    int dfs(int x, int y, vector<vector<int>>& matrix) {
        if (path[x][y] != 0) return path[x][y];  // ä½¿ç”¨å·²ç»å­˜å‚¨çš„æ•°æ®
        
        path[x][y] = 1;  // åˆå§‹åŒ–ä¸º1
        for (int i = 0; i < 4; i++) {
            int nx = x + d[i], ny = y + d[i+1];
            if (nx >= 0 && nx < n && ny >= 0 && ny < m && matrix[nx][ny] > matrix[x][y]) 
                path[x][y] = max(path[x][y], dfs(nx, ny, matrix) + 1);
        }
        return path[x][y];
    }
    int longestIncreasingPath(vector<vector<int>>& matrix) {
        n = matrix.size(), m = matrix[0].size();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                res = max(res, dfs(i, j, matrix));
            }
        }
        return res;
    }
};
